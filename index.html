Implement a function that concatenates all numbers in a range into a string: (given the numbers 5 and 10, the result should be 5678910).

                    
                const joinNumbersRange = (firstNumber, lastNumber) => {
                    let result = '';
                    let i = firstNumber;
                    while (i <= lastNumber) {
                        result = `${result}${i}`;
                        i = i + 1;
                    }
                    return result;
                }
